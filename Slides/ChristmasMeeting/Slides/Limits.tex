\begin{frame}
	\frametitle{Statistical Interpretation}
	
	
	\li{$\zprime \rightarrow \ell \ell$ and $\wprime \rightarrow \ell\nu$ ($\ell = e/\mu$) searches currently set Bayesian limits using the Bayesian Analysis Toolkit (BAT). {\scriptsize{(ATLAS default recommendation is Frequentist limits with RooStats)}}  }
	
	\li {Obtaining limits with the BAT code can be time consuming ($\mathcal{O}$10\textsuperscript{5} CPU hours for 1000 pseudo-experiments).}
		\lii {Need to try out other methods (such as Frequentist limits) in case they could save time and resources.}
	
	\li {Frequentist limits would provide a comparison for our current limits and could facilitate future combinations with other searches which use similar tools (e.g. VV/VH).}
	
	\li{Started producing limits using pseudo-experiments using RooStats workspace.}
		\lii{Tests with asymptotic limits showed that PE are needed for high mass region where statistics drop.}
	
	\li{ Observe differences between RooStats and BAT without systematics first to look for unexpected behaviour.}
	%	\lii{Running on {\color{ATLASBlue}\underline{Statistics only}} for initial tests.}
	
	
\end{frame}
